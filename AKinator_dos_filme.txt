import java.util.Scanner;

public class QuestionarioFilmes {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("Pergunta 1: Os atores do filme são reais (humanos)?");
        System.out.println("1 - Sim (50 tons de cinza, Titanic, Invocação do mal)");
        System.out.println("2 - Não (Galinho Little, Alvin e os Esquilos)");

        int resposta1 = scanner.nextInt();

        if (resposta1 == 1) {
            System.out.println("Pergunta 2: Tem assombração?");
            System.out.println("1 - Sim (Invocação do mal)");
            System.out.println("2 - Não (50 tons de cinza, Titanic)");

            int resposta2 = scanner.nextInt();

            if (resposta2 == 1) {
                System.out.println("Resposta: O filme é 'Invocação do mal'.");
            } else if (resposta2 == 2){
                System.out.println("Resposta: O filme é '50 tons de cinza' ou 'Titanic'.");
            }


        } else if (resposta1 == 2) {
            System.out.println("Pergunta 2b: É uma ficção?");
            System.out.println("1 - Sim (Alvin e os Esquilos)");
            System.out.println("2 - Não (Galinho Little)");

            int resposta2b = scanner.nextInt();

            if (resposta2b == 1) {
                System.out.println("Resposta: O filme é 'Alvin e os Esquilos'.");
            } else if (resposta2b == 2) {
                System.out.println("Resposta: O filme é 'Galinho Little'.");
            }
        }

        scanner.close();
    }
}
